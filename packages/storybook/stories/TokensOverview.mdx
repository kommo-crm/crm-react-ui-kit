</ReactMarkdown>

<ReactMarkdown>
  {i18n.t(

  )}
</ReactMarkdown>

```css
@import '@kommo-crm/crm-tokens/css/primitives';
```

<ReactMarkdown>
  {i18n.t(
    '**Semantic tokens** are theme-aware variables. Import the light theme, dark theme, or both, depending on your setup:'
  )}
</ReactMarkdown>

```css
@import '@kommo-crm/crm-tokens/css/semantic/light';
@import '@kommo-crm/crm-tokens/css/semantic/dark';
```

<ReactMarkdown>
  {i18n.t(
    'After importing the tokens, use the CSS custom properties anywhere you need:'
  )}
</ReactMarkdown>

```css
.button {
  background: var(--color-light-blue-500);
  border-radius: var(--border-radius-default);
}
```

<ReactMarkdown>{'## ' + i18n.t('JavaScript / TypeScript')}</ReactMarkdown>

<ReactMarkdown>
  {i18n.t(
    "The JS export provides typed access to each token's resolved value and CSS variable name."
  )}
</ReactMarkdown>

**Primitives:**

```ts
import { color, borderRadius, fontSize } from '@kommo-crm/crm-tokens';

color.light.blue[500].value; // '#4c8bf7'
color.light.blue[500].cssVar; // '--color-light-blue-500'

borderRadius.default.value; // '4px'
fontSize.base.value; // '15px'
```

**Semantic tokens:**

```ts
import lightTokens from '@kommo-crm/crm-tokens/js/semantic/light';
import darkTokens from '@kommo-crm/crm-tokens/js/semantic/dark';

lightTokens['background']['default'].cssVar; // '--background-default'
```

<ReactMarkdown>{i18n.t('Every token has the following shape:')}</ReactMarkdown>

```ts
interface Token {
  value: string; // resolved CSS value
  cssVar: string; // CSS custom property name (e.g. '--color-light-blue-500')
}
```
